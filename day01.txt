算法1：剑指Offer
题目：我们可以用2*1的小矩形横着或者竖着去覆盖更大的矩形。请问用n个2*1的小矩形无重叠地覆盖一个2*n的大矩形，总共有多少种方法？
{代码}
public class Solution {
    // 根据规律可以知道: 这是个类似斐波那契数列，f1 = 1，f2 = 2
    public int RectCover(int target) {
        int f1 = 0;
        int f2 = 1;
        int f =0;
        for(int i=0; i<target; ++i){
            f = f1 +f2;
            f1 = f2;
            f2 = f;
        }
        return f;
    }
}
算法2：LeetCode
题目：Given a binary tree, find its minimum depth.The minimum depth is the number of nodes along the shortest path from the root node down to the nearest leaf node.
{代码}
/**
 * Definition for binary tree
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
// 主要是理解题意。我一开始理解成了，求最小带权路径的所在的节点个数
// 题意是：求最短路径，也就是根节点到叶子节点得最短路径（节点数最少）
public class Solution {
    public int run(TreeNode root) {
        int min = 0; 
        if(root == null)// 为null，则返回 0
            return 0;
        int l = run(root.left); // 左子树的最短路径节点数
        int r = run(root.right);// 右子树的最短路径节点数
        
        if(l == 0 || r == 0)// 单分支
            min =  l + r + 1;
        else min  = (l>r? r:l) + 1; // 左右子树的路径大小比较，谁小选谁
            
        return min;
    }
}
