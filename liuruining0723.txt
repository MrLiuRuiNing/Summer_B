算法：
题目：合并两个升序链表
public class ListNode {
	    int val;
	    ListNode next = null;

	    ListNode(int val) {
	        this.val = val;
	    }
	}
	public ListNode Merge(ListNode list1,ListNode list2) {
	       if(list1 == null) {
	           return list2;
	       } 
	       if(list2 == null) {
	           return list1;
	       } 
	       ListNode s = null;
	        if(list1.val < list2.val) {
	             s = list1;
	           s.next =  Merge(list1.next, list2);
	        } else if(list1.val >= list2.val) {
	            s = list2;
	            s.next = Merge(list1, list2.next);
	        }
	        return s;
	    }
